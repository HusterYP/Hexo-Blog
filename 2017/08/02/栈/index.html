前言

数据结构中栈的相关操作，顺序栈，链栈

栈

栈是一种建立在线性表上的操作，是限制在表的同一端进行插入删除的
线性表，允许插入和删除的一端叫栈顶，固定端为栈底，分为顺序栈，
链栈；基本特点是先进后出.

顺序栈

顺序栈建立在线性表的顺序存储上，基本思想差不多

数据描述

#define ERROR -1 
#define OK 1
#define INITSIZE 10  //初始长度
#define INCREAMENT 2 //空间不够时每次增加的长度
#define Elemtype int

typedef struct
{
    Elemtype *elem;
    int top;    //栈顶指针，永远指向栈顶元素后面一个位置
    int stacksize;
}SeqStack;

初始化

int Init(SeqStack *S)    //常用一个返回值来判断是否成功
{
    S->elem = (Elemtype*)malloc(sizeof(Elemtype)*INITSIZE);
    if (!S->elem)    //判断是否动态分配成功
        return ERROR;
    S->top = 0;        //为0表示空栈
    S->stacksize = INITSIZE;
    return OK;
}

压栈

int Push(SeqStack *S, Elemtype e)
{
    Elemtype* temp;
    int newspaceSize;
    if (S->top >= S->stacksize - 1)    //若空间不足
    {
        temp = (Elemtype*)realloc(S->elem, sizeof (Elemtype)*(S->stacksize + INCREAMENT));  //见注
        if (!temp)
            return ERROR;
        S->elem = temp;
        S->stacksize += INCREAMENT;
    }
    S->elem[S->top] = e;
    S->top++;    //栈顶指针后移
    return OK;
}

注 ：关于realloc的使用可见realloc用法

弹栈

int Pop(SeqStack* S, Elemtype *e)
{
    if (S->top <= 0)        //是否空栈
        return ERROR;
    *e = S->elem[S->top - 1];     
    S->top--;
    return OK;
}

取栈顶元素

int GetTop(SeqStack* S, Elemtype* e)
{
    if (S->top <= 0)
        return ERROR;
    *e = S->elem[S->top - 1];
    return OK;
}

判断栈空

int StackEmpty(SeqStack *S)
{
    if (S->top <= 0)
        return 1;
    return 0;
}

清空栈

将栈顶指针置0，原来栈中存储的数据视为随机值，即清空

int ClearStack(SeqStack* S)
{
    S->top = 0;
    return OK;
}

链栈

链栈建立在带头结点的单链表基础上

数据描述

#define ERROR -1
#define OK 1
#define Elemtype int

typedef struct node
{
    Elemtype data;
    struct node *next;
}ListStack;

初始化

ListStack* Init()
{
    ListStack* S;
    S = (ListStack*)malloc(sizeof(ListStack));
    S->next = NULL;
    return S;
}

压栈

int Push(ListStack* S, Elemtype e)
{
    ListStack* p;
    p = (ListStack*)malloc(sizeof(ListStack));
    p->data = e;
    p->next = S->next;
    S->next = p;
    return OK;
}

弹栈

int Pop(ListStack* S, Elemtype* e)
{
    ListStack* p;
    if (S->next == NULL)
        return ERROR;
    p = S->next;
    *e = p->data;
    S->next = p->next;
    free(p);
    return OK;
}

取栈顶

int GetTop(ListStack* S, Elemtype* e)
{
    if (S->next == NULL)
        return ERROR;
    *e = S->next->data;
    return OK;
}

清空栈

int ClearStack(ListStack* S)
{
    ListStack* p, *q;
    for (p = S; p != NULL; p = q)
    {
        q = p->next;
        free(p);
    }
    return OK;
}

共享栈(双端栈)

基于顺序栈；两个顺序栈共享一个一维数据空间，两个栈的栈底分别在
共享空间的两端，栈顶向数据空间中间延伸；共享栈的数据元素个数半
不能超过M,超过就会溢出；应用软件的撤销和恢复操作就是用的共享栈

数据描述

M表示最大存储空间；

#define ERROR -1
#define OK 1
#define Elemtype int
#define M 20

typedef struct
{
    Elemtype elem[M];
    int top1;
    int top2;
}SharedStack;

初始化

即将两个栈置空；top1=-1表示一号栈为空，top2=M表示二
号栈为空；(因为top1=-1的时候表示空，所以一号栈压栈的时候要先
加一，再压栈;二号栈要先指针减一再压栈)

int Init(SharedStack* S)
{
    S->top1 = -1;
    S->top2 = M;
    return OK;
}

压栈

int Push(SharedStack* S, Elemtype e, int i)
{
    if (S->top1 + 1 == S->top2)  //判断栈满
        return ERROR;
    switch (i)      //i表示压哪个栈
    {
    case 1:
        S->top1++;
        S->elem[S->top1] = e;
        break;
    case 2:
        S->top2--;
        S->elem[S->top2] = e;
        break;
    default:
        return ERROR;
    }
    return OK;
}

弹栈

int Pop(SharedStack* S, Elemtype* e ,int i)
{
    switch (i)
    {
    case 1:
        if (S->top1 == -1)
            return ERROR;
        *e = S->elem[S->top1];
        S->top1--;
        break;
    case 2:
        if (S->top2 == M)
            return ERROR;
        *e = S->elem[S->top2];
        S->top2++;
        break;
    default:
        break;
    }
    return OK;
}

动态栈

考虑到共享栈的空间不能改变，某些场合不方便，因此将结构体中的数
组换成指针，使用动态分配；当栈满的时候用realloc先扩容，再压
栈;基本操作类似，只是换成了对存储空间的动态分配而已

多链栈

适合于需要多个栈解决问题；可基于顺序栈，也可基于链栈，下面讨论
基于链栈的情况

数据描述

每个单链栈都有自己的数据区

#define ERROR -1
#define OK 1
#define Elemtype int
#define M 5

typedef struct node  //每个单链栈节点
{
    Elemtype data;
    struct node* next;
}ListStackNode;

typedef struct     //储存多链栈的信息
{ 
    ListStackNode *top[M];     //存储每个单链栈的头指针的指针数组
    int stacknum;    //当前单链栈的个数
};

初始化

int Init(MulListStack* S)
{
    for (int i = 0; i < M; i++)
    {
        S->top[i] = (ListStackNode*)malloc(sizeof(ListStackNode));
        if (S->top[i] == NULL)
            return ERROR;
        S->top[i]->next = NULL; // 不要忘了将next域置空
    }
    S->stacknum = M;
    return OK;
}

压栈

int Push(MulListStack* S, Elemtype e, int i)
{
    ListStackNode* p;
    if (i<1 || i>S->stacknum)  //注意这里的i表示的是压第几个栈，而不是数组下标
        return ERROR;
    p = (ListStackNode*)malloc(sizeof(ListStackNode));
    if (p == NULL)
        return ERROR;
    p->data = e;
    p->next = S->top[i - 1]->next;
    S->top[i - 1]->next = p;
    return OK;
}

弹栈

int Pop(MulListStack* S, int i, Elemtype *e)
{
    ListStackNode* p;
    if (i<1 || i>S->stacknum)
        return ERROR;
    p = S->top[i - 1]->next;
    if (p == NULL)    //判断栈空
        return ERROR;
    S->top[i]->next = p->next;
    *e = p->data;
    free(p);
    return OK;
}
