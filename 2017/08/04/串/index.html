int Concat(SString T, SString S1, SString S2)
{
    int i;
    if (S1[0] + S2[0] <= INITSIZE)
    {
        for (i = 1; i <= S1[0]; i++)
            T[i] = S1[i];
        for (i = 1; i < S2[0]; i++)
            T[i + S1[0]] = S2[i];
        T[0] = S1[0] + S2[0];
        return TRUE;
    }
    else if (S1[0] < INITSIZE)
    {
        for (i = 1; i < S1[0]; i++)
            T[i] = S1[i];
        for (i = S1[0] + 1; i <= INITSIZE; i++)
            T[i] = S2[i - S1[0]];
        T[0] = INITSIZE;
        return FALSE;
    }
    else
    {
        for (i = 0; i <= INITSIZE; i++) //这里的目的是仅截取S1，所以从i=0开始
            T[i] = S1[i];
        return FALSE;
    }
    return FALSE;
}
