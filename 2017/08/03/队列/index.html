前言

队列相关操作

队列

先进先出，允许插入端为队尾，允许删除端为队头；为了避免溢出，常使用循环队列的形式

循环队列

数据结构

顺序存储结构

#define ERROR -1
#define OK 1
#define Elemtype int
#define INITSIZE 20

typedef struct
{
    Elemtype *elem;
    int front;
    int rear;
    int QueueSize;
}SeqQueue;

初始化

int Init(SeqQueue* Q)
{
    Q->elem = (Elemtype*)malloc(sizeof(Elemtype)*INITSIZE);
    if (!Q->elem)
        return ERROR;
    Q->QueueSize = INITSIZE;
    Q->front = Q->rear = 0;        //这里只要二者相等就行了，其实不需要等于0
    return OK;
}

入队操作

注音入队判断队满的情况，当队尾指针之后一个位置为队头的话即队满，但是其实这时候还有一个位置可以存储，准确的说判断队满是要队尾指针增加到和队头指针相等，但是由于判断队空时也是这样判断的，为了避免冲突，所以队满的判断要像下面这样咯

int EnQueue(SeqQueue* Q, Elemtype e)
{
    if ((Q->rear + 1) % Q->QueueSize == Q->front)
        return ERROR;
    Q->elem[Q->rear] = e;
    Q->rear = (Q->rear + 1) % Q->QueueSize;
    return OK;
}

出队操作

当队尾指针和队头指针相同时，判断队空

int DeQueue(SeqQueue* Q, Elemtype* e)
{
    if (Q->rear == Q->front)
        return ERROR;
    *e = Q->elem[Q->front];
    Q->front = (Q->front + 1) % Q->QueueSize;
    return OK;
}

求队长

int QueLength(SeqQueue* Q)
{
    return Q->rear - Q->front + Q->QueueSize;
}

动态循环队列

动态循环队列是在队满是通过realloc动态扩容；当存储完数据，发现数据空间有剩余时，选择性的对空间进行回收

数据结构

#define QUEUE_INITSIZE 20  //初始长度
#define QUEUE_INCREAMENT 5     //增加长度
#define QUEUE_PLANSIZE 10     //回收的空闲预定值
#define QUEUE_PERCENT 0.5    //回收的空闲预定比例
#define Elemtype int
typedef struct
{
    Elemtype* elem;
    int rear;
    int front;
    int QueueSize;
}SeqQueue;

链队列

基于链式存储结构，适合数据元素变动比较大的时候

#define ERROR -1
#define OK 1
#define Elemtype int

typedef struct node
{
    Elemtype data;
    struct node* next;
}QueueNode;
typedef struct
{
    QueueNode *front;
    QueueNode *rear;
}LinkQueue;

初始化

初始化时只需要分配头指针的空间即可，不需要分配尾指针的空间

int Init(LinkQueue* Q)
{
    Q->front = (QueueNode*)malloc(sizeof(QueueNode));
    Q->front->next = NULL;
    Q->rear = Q->front;
    return OK;
}

##入队

int EnQueue(LinkQueue* Q, Elemtype e)
{
    QueueNode* p;
    p = (QueueNode*)malloc(sizeof(QueueNode));
    p->data = e;
    p->next = NULL;        //记住置空
    Q->rear->next = p;    
    Q->rear = p;
    return OK;
}

出队

注意出队时，因为尾指针没有分配空间，所以下面的p又可能和尾指针同指向，所以在free(p)的时候还要判断是否和尾指针同指向

int DeQueue(LinkQueue* Q, Elemtype* e)
{
    QueueNode* p;
    if (Q->rear == Q->front)
        return ERROR;
    p = Q->front->next;
    *e = p->data;
    Q->front->next = p->next;
    if (p == Q->rear)     //判断是否和尾指针同指向
        Q->rear = Q->front;
    free(p);
    return OK;
}

循环多队列

初始化

int Init(MultyQueue* Q)
{
    for (int i = 0; i < N; i++)
        Q->f[i].front = Q->f[i].rear = 0;
    return OK;
}

入队

int EnQueue(MultyQueue* Q, int i, Elemtype e)
{
    if (i<1 || i>N)        //注意i表示的是第i条队列
        return ERROR;
    if ((Q->f[i - 1].rear + 1) % M == Q->f[i - 1].front)   //判断队满
        return ERROR;
    Q->queue[i - 1][Q->f[i - 1].rear] = e;
    Q->f[i - 1].rear = (Q->f[i - 1].rear + 1) % M;
    return OK;
}

出队

int EnQueue(MultyQueue* Q, int i, Elemtype e)
{
    if (i<1 || i>N)
        return ERROR;
    if ((Q->f[i - 1].rear + 1) % M == Q->f[i - 1].front)
        return ERROR;
    Q->queue[i - 1][Q->f[i - 1].rear] = e;
    Q->f[i - 1].rear = (Q->f[i - 1].rear + 1) % M;
    return OK;
}
